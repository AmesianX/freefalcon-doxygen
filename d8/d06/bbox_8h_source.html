429 bbox3::ClipStatus
430 bbox3::clipstatus(const matrix44& viewProjection) const
431 {
432  int andFlags = 0xffff;
433  int orFlags = 0;
434  int i;
435  static vector4 v0;
436  static vector4 v1;
437 
438  for (i = 0; i < 8; i++)
439  {
440  int clip = 0;
441  v0.w = 1.0f;
442 
443  if (i & 1) v0.x = this->vmin.x;
444  else v0.x = this->vmax.x;
445 
446  if (i & 2) v0.y = this->vmin.y;
447  else v0.y = this->vmax.y;
448 
449  if (i & 4) v0.z = this->vmin.z;
450  else v0.z = this->vmax.z;
451 
452  v1 = viewProjection * v0;
453 
454  if (v1.x < -v1.w) clip |= ClipLeft;
455  else if (v1.x > v1.w) clip |= ClipRight;
456 
457  if (v1.y < -v1.w) clip |= ClipBottom;
458  else if (v1.y > v1.w) clip |= ClipTop;
459 
460  if (v1.z < -v1.w) clip |= ClipFar;
461  else if (v1.z > v1.w) clip |= ClipNear;
462 
463  andFlags &= clip;
464  orFlags |= clip;
465  }
466 
467  if (0 == orFlags) return Inside;
468  else if (0 != andFlags) return Outside;
469  else return Clipped;

480 inline bool bbox3::intersect(const line3& line, vector3& ipos) const
481 {
482  // Handle special case for start point inside box
483  if (line.b.x >= vmin.x && line.b.y >= vmin.y && line.b.z >= vmin.z &&
484  line.b.x <= vmax.x && line.b.y <= vmax.y && line.b.z <= vmax.z)
485  {
486  ipos = line.b;
487  return true;
488  }
489 
490  // Order planes to check, closest three only
491  int plane[3];
492 
493  if (line.m.x > 0) plane[0] = 0;
494  else plane[0] = 1;
495 
496  if (line.m.y > 0) plane[1] = 2;
497  else plane[1] = 3;
498 
499  if (line.m.z > 0) plane[2] = 4;
500  else plane[2] = 5;
501 
502  for (int i = 0; i < 3; ++i)
503  {
504  switch (plane[i])
505  {
506  case 0:
507  if (isect_const_x(vmin.x, line, ipos) && pip_const_x(ipos)) return true;
508 
509  break;
510 
511  case 1:
512  if (isect_const_x(vmax.x, line, ipos) && pip_const_x(ipos)) return true;
513 
514  break;
515 
516  case 2:
517  if (isect_const_y(vmin.y, line, ipos) && pip_const_y(ipos)) return true;
518 
519  break;
520 
521  case 3:
522  if (isect_const_y(vmax.y, line, ipos) && pip_const_y(ipos)) return true;
523 
524  break;
525 
526  case 4:
527  if (isect_const_z(vmin.z, line, ipos) && pip_const_z(ipos)) return true;
528

530 
531  case 5:
532  if (isect_const_z(vmax.z, line, ipos) && pip_const_z(ipos)) return true;

559 int bbox3::intersect(bbox3 box)
560 {
561  int and_code = 0xffff;
562  int or_code = 0;
563  int cx, cy, cz;
564  cx = line_test(vmin.x, vmax.x, box.vmin.x, box.vmax.x);
565  and_code &= cx;
566  or_code |= cx;
567  cy = line_test(vmin.y, vmax.y, box.vmin.y, box.vmax.y);
568  and_code &= cy;
569  or_code |= cy;
570  cz = line_test(vmin.z, vmax.z, box.vmin.z, box.vmax.z);
571  and_code &= cz;
572  or_code |= cz;
573 
574  if (or_code == 0) return OUTSIDE;
575  else if (and_code != 0)
576  {
577  return and_code;
578  }
579  else
580  {
581  // only if all test produced a non-outside result,
582  // an intersection has occured
583  if (cx && cy && cz) return CLIPS;
584  else return OUTSIDE;
585  }
