130 polar2::set(const vector3& vec)
131 {
132  double dTheta = acos(vec.y);
133 
134  // build a normalized 2d vector of the xz component
135  vector2 v2(vec.x, vec.z);
136  v2.norm();
137 
138  // adjust dRho based on the quadrant we are in
139  double dRho;
140 
141  if ((v2.x >= 0.0f) && (v2.y >= 0.0f))
142  {
143  // quadrant 1
144  dRho = acos(v2.x);
145  }
146  else if ((v2.x < 0.0f) && (v2.y >= 0.0f))
147  {
148  // quadrant 2
149  dRho = acos(v2.y) + _deg2rad(90.0f);
150  }
151  else if ((v2.x < 0.0f) && (v2.y < 0.0f))
152  {
153  // quadrant 3
154  dRho = acos(-v2.x) + _deg2rad(180.0f);
155  }
156  else
157  {
158  // quadrant 4
159  dRho = acos(-v2.y) + _deg2rad(270.0f);
160  }
161 
162  this->theta = (float) dTheta;
163  this->rho = (float) dRho;
164 }
