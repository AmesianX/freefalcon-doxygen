21  * AMD3DX.H
22  *
23  * MACRO FORMAT
24  * ============
25  * This file contains inline assembly macros that
26  * generate AMD-3D instructions in binary format.
27  * Therefore, C or C++ programmer can use AMD-3D instructions
28  * without any penalty in their C or C++ source code.
29  *
30  * The macro's name and format conventions are as follow:
31  *
32  *
33  * 1. First argument of macro is a destination and
34  * second argument is a source operand.
35  * ex) _asm PFCMPEQ (mm3, mm4)
36  * | |
37  * dst src
38  *
39  * 2. The destination operand can be m0 to m7 only.
40  * The source operand can be any one of the register
41  * m0 to m7 or _eax, _ecx, _edx, _ebx, _esi, or _edi
42  * that contains effective address.
43  * ex) _asm PFRCP (MM7, MM6)
44  * ex) _asm PFRCPIT2 (mm0, mm4)
45  * ex) _asm PFMUL (mm3, _edi)
46  *
47  * 3. The prefetch(w) takes one src operand _eax, ecx, _edx,
48  * _ebx, _esi, or _edi that contains effective address.
49  * ex) _asm PREFETCH (_edi)
50  *
51  * For WATCOM C/C++ users, when using #pragma aux instead if
52  * _asm, all macro names should be prefixed by a p_ or P_.
53  * Macros should not be enclosed in quotes.
54  * ex) p_pfrcp (MM7,MM6)
55  *
56  * NOTE: Not all instruction macros, nor all possible
57  * combinations of operands have been explicitely
58  * tested. If any errors are found, please report
59  * them.
60  *
61  * EXAMPLE
62  * =======
63  * Following program doesn't do anything but it shows you
64  * how to use inline assembly AMD-3D instructions in C.
65  * Note that this will only work in flat memory model which
66  * segment registers cs, ds, ss and es point to the same
67  * linear address space total less than 4GB.
68  *
69  * Used Microsoft VC++ 5.0
70  *
71  * #include <stdio.h>
72  * #include "amd3d.h"
73  *
74  * void main ()
75  * {
76  * float x = (float)1.25;
77  * float y = (float)1.25;
78  * float z, zz;
79  *
80  * _asm {
81  * movd mm1, x
82  * movd mm2, y
83  * pfmul (mm1, mm2)
84  * movd z, mm1
85  * femms
86  * }
87  *
88  * printf ("value of z = %f\n", z);
89  *
90  * //
91  * // Demonstration of using the memory instead of
92  * // multimedia register
93  * //
94  * _asm {
95  * movd mm3, x
96  * lea esi, y // load effective address of y
97  * pfmul (mm3, _esi)
98  * movd zz, mm3
99  * femms
100  * }
101  *
102  * printf ("value of zz = %f\n", zz);
103  * }
104  *
105  * #pragma aux EXAMPLE with WATCOM C/C++ v11.x
106  * ===========================================
107  *
108  * extern void Add(float *__Dest, float *__A, float *__B);
109  * #pragma aux Add = \
110  * p_femms \
111  * "movd mm6,[esi]" \
112  * p_pfadd(mm6,_edi) \
113  * "movd [ebx],mm6" \
114  * p_femms \
115  * parm [ebx] [esi] [edi];
116  *
117  **********************************************************/
