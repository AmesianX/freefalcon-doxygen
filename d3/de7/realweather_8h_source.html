670 inline void RotatePoint(Tpoint *t, float p, float r, float y)
671 {
672  Trotation tRot;
673  float costha, sintha, cosphi, sinphi, cospsi, sinpsi;
674 
675  costha = Cos(p);
676  sintha = Sin(p);
677  cosphi = Cos(r);
678  sinphi = Sin(r);
679  cospsi = Cos(y);
680  sinpsi = Sin(y);
681 
682  tRot.M11 = cospsi * costha;
683  tRot.M21 = sinpsi * costha;
684  tRot.M31 = -sintha;
685 
686  tRot.M12 = -sinpsi * cosphi + cospsi * sintha * sinphi;
687  tRot.M22 = cospsi * cosphi + sinpsi * sintha * sinphi;
688  tRot.M32 = costha * sinphi;
689 
690  tRot.M13 = sinpsi * sinphi + cospsi * sintha * cosphi;
691  tRot.M23 = -cospsi * sinphi + sinpsi * sintha * cosphi;
692  tRot.M33 = costha * cosphi;
693 
694  register float s_x = tRot.M11 * t->x + tRot.M12 * t->y + tRot.M13 * t->z;
695  register float s_y = tRot.M21 * t->x + tRot.M22 * t->y + tRot.M23 * t->z;
696  register float s_z = tRot.M31 * t->x + tRot.M32 * t->y + tRot.M33 * t->z;
697 
698  t->x = s_x;
699  t->y = s_y;
700  t->z = s_z;
701 }
