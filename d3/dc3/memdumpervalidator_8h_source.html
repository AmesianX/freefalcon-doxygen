26  // The typedefs for the dumper and validator functions.
28  // The memory dumper function. The only parameter is a pointer to the
29  // block of memory. This function can output the memory data for the
30  // block any way it likes but it might be nice if it uses the same
31  // Debug CRT reporting mechanism that everything else in the runtime
32  // uses.
33  typedef void (*PFNMEMDUMPER)(const void *) ;
34  // The validator function. The first parameter is the memory block to
35  // validate and the second parameter is the context information passed
36  // to the ValidateAllBlocks function.

65  /*----------------------------------------------------------------------
66  FUNCTION : AddClientDV
67  DISCUSSION :
68  Adds a client block dumper and validator to the list. If the
69  dwValue field in the DVINFO structure is ZERO, then the next value in
70  the list is assigned. This means that the value returned must always be
71  passed to _malloc_dbg as the value of the client block.
72  If the value is set with CLIENT_BLOCK_VALUE, then a macro can be
73  used for the value to _malloc_dbg.
74  No, there is no corresponding remove function. Why possibly
75  introduce bugs in debugging code? Performance is a non issue when it
76  comes to finding errors.
77  PARAMETERS :
78  lpDVInfo - The pointer to the DVINFO structure.
79  RETURNS :
80  1 - The client block dumper and validator was properly added.
81  0 - The client block dumper and validator could not be added.
82  ----------------------------------------------------------------------*/
83  int BUGSUTIL_DLLINTERFACE __stdcall AddClientDV(LPDVINFO lpDVInfo);
84 
85  /*----------------------------------------------------------------------
86  FUNCTION : ValidateAllBlocks
87  DISCUSSION :
88  Checks all the memory allocated out of the local heap. Also goes
89  through all client blocks and calls the special validator function for
90  the different types of client blocks.
91  It is probably best to call this function with the VALIDATEALLBLOCKS
92  macro below.
93  PARAMETERS :
94  pContext - The context information that will be passed to each
95  call to the validator function.
96  RETURNS :
97  None.
98  ----------------------------------------------------------------------*/
99  void BUGSUTIL_DLLINTERFACE __stdcall
100  ValidateAllBlocks(void * pContext) ;
