1 /*----------------------------------------------------------------------
2  John Robbins - Microsoft Systems Journal Bugslayer Column
3 ------------------------------------------------------------------------
4 
5  This is a paper thin layer over the IMAGEHLP.DLL symbol engine which
6 helps hide some of the differences between the original NT4.0 version
7 and the November Platform SDK and later version.
8 
9  Note that this class only wraps those functions that take the
10 unique HANDLE value. Other IMAGEHLP.DLL symbol engine functions are
11 global in scope so I did not wrap them with this class.
12 
13  While IMAGEHLP.DLL has a version API, ImagehlpApiVersion and
14 ImagehlpApiVersionEx, it is very weird. No matter which of the many
15 builds of IMAGEHLP.DLL I checked, they all returned major version 4,
16 minor version 0, revision 5. This was even the case with the new NT5
17 IMAGEHLP.DLL. Whatever. What I did instead was to just look for the
18 line specific exported functions with GetProcAddress. If they are
19 there, then they can be used. To see if the line information functions
20 are supported, call the classes CanDoSourceLines function.
21 
22  I was able to take care of almost all the differences between the
23 different versions here. The one difference is that the
24 IMAGEHLP_DEFERRED_SYMBOL_LOAD structure is a larger size in the November
25 Platform SDK version.
26 
27 ------------------------------------------------------------------------
28 Compilation Defines:
29 
30 FLEXIBLE_SYMBOLENGINE - Define this to alway have the CSymbolEngine
31  class use the GetProcAddress method of
32  determining if the IMAGEHLP.DLL in memory
33  supports the new source and line functions.
34 
35 DO_NOT_WORK_AROUND_SRCLINE_BUG - Define this to NOT work around the
36  SymGetLineFromAddr bug where PDB file
37  lookups fail after the first lookup.
38 ----------------------------------------------------------------------*/

110  Whenever I ran the release build, it always crashed because the
111  stack was not restoring the registers when certain functions in this
112  class were called. Looking at the disassembly, I kept seeing
113  instructions like ADD ESP,10h after certain calls. This was driving
114  me nuts until it dawned on me that the extra ADD after the calls
115  were cleaning up the stack for CDECL calls but that Windows API
116  calls are all STDCALL. I had a calling convention mismatch!

119  problems if you do something like this. Watch those calling

315  // The worker function used by the previous two functions.
316  BOOL GetInMemoryFileVersion(LPCTSTR szFile ,
317  DWORD & dwMS ,
318  DWORD & dwLS)
319  {
320  HMODULE hInstIH = GetModuleHandle(szFile) ;
321 
322  // Get the full filename of the loaded version.
323  TCHAR szImageHlp[ MAX_PATH ] ;
324  GetModuleFileName(hInstIH , szImageHlp , MAX_PATH) ;
325 
326  dwMS = 0 ;
327  dwLS = 0 ;
328 
329  // Get the version info size.
330  DWORD dwVerInfoHandle ;
331  DWORD dwVerSize ;
332 
333  dwVerSize = GetFileVersionInfoSize(szImageHlp ,
334  &dwVerInfoHandle) ;
335 
336  if (0 == dwVerSize)
337  {
338  return (FALSE) ;
339  }
340 
341  // Got the version size, now get the version info.
342  LPVOID lpData = (LPVOID)new TCHAR [ dwVerSize ] ;
343 
344  if (FALSE == GetFileVersionInfo(szImageHlp ,
345  dwVerInfoHandle ,
346  dwVerSize ,
347  lpData))
348  {
349  delete [] lpData ;
350  return (FALSE) ;
351  }
352 
353  VS_FIXEDFILEINFO * lpVerInfo ;
354  UINT uiLen ;
355  BOOL bRet = VerQueryValue(lpData ,
356  _T("\\") ,
357  (LPVOID*)&lpVerInfo ,
358  &uiLen) ;
359 
360  if (TRUE == bRet)
361  {
362  dwMS = lpVerInfo->dwFileVersionMS ;
363  dwLS = lpVerInfo->dwFileVersionLS ;
364  }
365 
366  delete [] lpData ;
367 
368  return (bRet) ;

496  BOOL SymGetLineFromAddr(IN DWORD dwAddr ,
497  OUT PDWORD pdwDisplacement ,
498  OUT PIMAGEHLP_LINE Line)
499  {
500  // Holds the function pointer.
501  PFNSYMGETLINEFROMADDR pfnSGLFA ;
502 
503 #ifdef SYMENG_EXTRAWORK
504 
505  if (NULL == m_pfnSymGetLineFromAddr)
506  {
507  return (FALSE) ;
508  }
509  else
510  {
511  pfnSGLFA = m_pfnSymGetLineFromAddr ;
512  }
513 
514 #else
515  // Just grab it directly.
516  pfnSGLFA = ::SymGetLineFromAddr ;
517 #endif // SYMENG_EXTRAWORK
518 
519 #ifdef DO_NOT_WORK_AROUND_SRCLINE_BUG
520  // Just return whatever the main function returned.
521  return (pfnSGLFA(m_hProcess ,
522  dwAddr ,
523  pdwDisplacement ,
524  Line)) ;
525 
526 #else
527  // The problem is that the symbol engine only finds those source
528  // line addresses (after the first lookup) that fall exactly on
529  // a zero displacement. I will walk backwards 100 bytes to
530  // find the line and return the proper displacement.
531  DWORD dwTempDis = 0 ;
532 
533  while (FALSE == pfnSGLFA(m_hProcess ,
534  dwAddr - dwTempDis ,
535  pdwDisplacement ,
536  Line))
537  {
538  dwTempDis += 1 ;
539 
540  if (100 == dwTempDis)
541  {
542  return (FALSE) ;
543  }
544  }
545 
546  // It was found and the source line information is correct so
547  // change the displacement if it was looked up multiple times.
548  if (0 != dwTempDis)
549  {
550  *pdwDisplacement = dwTempDis ;
551  }
552 
553  return (TRUE) ;
554 #endif // DO_NOT_WORK_AROUND_SRCLINE_BUG
555  }
