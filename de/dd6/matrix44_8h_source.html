114  // Convert orientation of 4x4 matrix into quaterion, 4x4 matrix must not be scaled!
115  quaternion GetQuaternion() const
116  {
117  float qa[4];
118  float tr = m[0][0] + m[1][1] + m[2][2];
119 
120  if (tr > 0.0f)
121  {
122  float s = sqrt(tr + 1.0f);
123  qa[3] = s * 0.5f;
124  s = 0.5f / s;
125  qa[0] = (m[1][2] - m[2][1]) * s;
126  qa[1] = (m[2][0] - m[0][2]) * s;
127  qa[2] = (m[0][1] - m[1][0]) * s;
128  }
129  else
130  {
131  int i, j, k, nxt[3] = {1, 2, 0};
132  i = 0;
133 
134  if (m[1][1] > m[0][0]) i = 1;
135 
136  if (m[2][2] > m[i][i]) i = 2;
137 
138  j = nxt[i];
139  k = nxt[j];
140  float s = sqrt((m[i][i] - (m[j][j] + m[k][k])) + 1.0f);
141  qa[i] = s * 0.5f;
142  s = 0.5f / s;
143  qa[3] = (m[j][k] - m[k][j]) * s;
144  qa[j] = (m[i][j] + m[j][i]) * s;
145  qa[k] = (m[i][k] + m[k][i]) * s;
146  }
147 
148  quaternion q(qa[0], qa[1], qa[2], qa[3]);
149  return q;
150  }

286  /* Inverts a 4x4 matrix consisting of a 3x3 rotation matrix and a translation (eg. everything
287  that has [0,0,0,1] as the rightmost column) MUCH cheaper then a real 4x4 inversion */
288  void InvertSimple()
289  {
290  float s = Det();
291 
292  if (s == 0.0f) return;
293 
294  s = 1.0f / s;
295  this->Set(
296  s * ((m22 * m33) - (m23 * m32)),
297  s * ((m32 * m13) - (m33 * m12)),
298  s * ((m12 * m23) - (m13 * m22)),
299  0.0f,
300  s * ((m23 * m31) - (m21 * m33)),
301  s * ((m33 * m11) - (m31 * m13)),
302  s * ((m13 * m21) - (m11 * m23)),
303  0.0f,
304  s * ((m21 * m32) - (m22 * m31)),
305  s * ((m31 * m12) - (m32 * m11)),
306  s * ((m11 * m22) - (m12 * m21)),
307  0.0f,
308  s * (m21 * (m33 * m42 - m32 * m43) + m22 * (m31 * m43 - m33 * m41) + m23 * (m32 * m41 - m31 * m42)),
309  s * (m31 * (m13 * m42 - m12 * m43) + m32 * (m11 * m43 - m13 * m41) + m33 * (m12 * m41 - m11 * m42)),
310  s * (m41 * (m13 * m22 - m12 * m23) + m42 * (m11 * m23 - m13 * m21) + m43 * (m12 * m21 - m11 * m22)),
311  1.0f);
312  }

314  // Optimized multiplication, assumes that M14==M24==M34==0 AND M44==1
315  void MultSimple(const matrix44& m1)
316  {
317  int i;
318 
319  for (i = 0; i < 4; i++)
320  {

330  m[3][1] += m1.m[3][1];

605  void Rotate(const vector3& vec, float a)
606  {
607  vector3 v(vec);
608  v.Normalize();
609  float sa = (float) sinf(a);
610  float ca = (float) cosf(a);

613  rotM.m11 = ca + (1.0f - ca) * v.x * v.x;
614  rotM.m12 = (1.0f - ca) * v.x * v.y - sa * v.z;
615  rotM.m13 = (1.0f - ca) * v.z * v.x + sa * v.y;

617  rotM.m22 = ca + (1.0f - ca) * v.y * v.y;
618  rotM.m23 = (1.0f - ca) * v.y * v.z - sa * v.x;
619  rotM.m31 = (1.0f - ca) * v.z * v.x - sa * v.y;

657  void SetView(vector3& vFrom, vector3& vAt, vector3& vWorldUp)
658  {
659  // Get the z basis vector, which points straight ahead. This is the difference from the eyepoint to the lookat point.
660  vector3 vView = vAt - vFrom;
661  float fLength = vView.Length();
662 
663  if (fLength < 1e-6f) return;
664 
665  vView /= fLength; // Normalize the z basis vector
666 
667  // Get the dot product, and calculate the projection of the z basis vector onto the up vector. The projection is the y basis vector.
668  float fDotProduct = vWorldUp % vView;
669  vector3 vUp = vWorldUp - fDotProduct * vView;
670 
671  // If this vector has near-zero length because the input specified a bogus up vector, let's try a default up vector
672  if (1e-6f > (fLength = vUp.Length()))
673  {
674  vUp = vector3(0.f, 1.f, 0.f) - vView.y * vView;
675 
676  if (1e-6f > (fLength = vUp.Length())) // If we still have near-zero length, resort to a different axis.
677  {
678  vUp = vector3(0.f, 0.f, 1.f) - vView.z * vView;
679 
680  if (1e-6f > (fLength = vUp.Length())) return;
681  }
682  }
683 
684  vUp /= fLength; // Normalize the y basis vector
685 
686  // The x basis vector is found simply with the cross product of the y and z basis vectors
687  vector3 vRight = vUp * vView;
688 
689  // Start building the matrix. The first three rows contains the basis vectors used to rotate the view to point at the lookat point
690  Ident();
691  m[0][0] = vRight.x;
692  m[0][1] = vUp.x;
693  m[0][2] = vView.x;
694  m[1][0] = vRight.y;
695  m[1][1] = vUp.y;
696  m[1][2] = vView.y;
697  m[2][0] = vRight.z;
698  m[2][1] = vUp.z;
699  m[2][2] = vView.z;
700 
701  // Do the translation values (rotations are still about the eyepoint)
702  m[3][0] = -(vFrom % vRight);
703  m[3][1] = -(vFrom % vUp);
704  m[3][2] = -(vFrom % vView);
705  }
